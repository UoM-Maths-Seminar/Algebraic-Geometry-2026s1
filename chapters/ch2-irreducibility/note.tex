\section{Basic Definitions}

\begin{definition}[Affine Variety]
    An affine variety is the set of common zeros of a collection of polynomials in $\mathbb{A}^n$.
\end{definition}

\begin{example}
    The unit circle in $\mathbb{A}^2$ is given by $V(x^2 + y^2 - 1)$.
\end{example}

\section{Theorems and Proofs}

\begin{theorem}[Nullstellensatz]
    Let $I \subset k[x_1, \ldots, x_n]$ be an ideal. Then $I(V(I)) = \sqrt{I}$.
\end{theorem}

\begin{proof}
    The proof follows from the correspondence between ideals and algebraic sets.
\end{proof}
